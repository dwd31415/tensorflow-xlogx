    \label{simple_equation}
    \ell(x) =
    \left\{\begin{array}{lr}
        x \log (x) & \text{if } x > 0\\
        0 & \text{if } x = 0\\
      \end{array} \right.
\end{align*}

  \lim_{x \to 0} x \log(x) = 0
  \end{align*}
